48 bool AllWize_LoRaWAN::send(uint8_t *Data, uint8_t Data_Length, uint8_t Frame_Port) {
49 
50  // Define variables
51  uint8_t i;
52  uint8_t LoRaWAN_Data[64] = {0};
53  uint8_t LoRaWAN_Data_Length = 0;
54  uint8_t MIC[4] = {0};
55 
56  // Direction of frame is up
57  uint8_t Direction = 0x00;
58 
59  // Unconfirmed data up
60  // [7..5] MType (010 unconfirmed up, 100 confirmed up,...)
61  // [4..2] RFU
62  // [1..0] Major
63  uint8_t Mac_Header = 0x40;
64 
65  // Frame control
66  // [7] ADR
67  // [6] ADRACKReq
68  // [5] ACK
69  // [4] ClassB
70  // [3..0] FOptsLen
71  uint8_t Frame_Control = 0x00;
72 
73  // Make a copy of Data
74  uint8_t tmpData[Data_Length];
75  for (int i = 0; i < Data_Length; i++) {
76  tmpData[i] = Data[i];
77  }
78 
79  // Encrypt Data (data argument is overwritten in this function)
80  Encrypt_Payload(tmpData, Data_Length, _frame_counter, Direction);
81 
82  // MAC Header
83  LoRaWAN_Data[0] = Mac_Header;
84 
85  // MAC Payload - Frame Header
86  LoRaWAN_Data[1] = _devaddr[3];
87  LoRaWAN_Data[2] = _devaddr[2];
88  LoRaWAN_Data[3] = _devaddr[1];
89  LoRaWAN_Data[4] = _devaddr[0];
90  LoRaWAN_Data[5] = Frame_Control;
91  LoRaWAN_Data[6] = (_frame_counter & 0x00FF);
92  LoRaWAN_Data[7] = ((_frame_counter >> 8) & 0x00FF);
93 
94  // MAC Payload - Frame Port
95  LoRaWAN_Data[8] = Frame_Port;
96 
97  //Set Current package length
98  LoRaWAN_Data_Length = 9;
99 
100  // MAC Payload - Frame Payload
101  for(i = 0; i < Data_Length; i++) {
102  LoRaWAN_Data[LoRaWAN_Data_Length + i] = tmpData[i];
103  }
104  LoRaWAN_Data_Length += Data_Length;
105 
106  // Calculate MIC
107  Calculate_MIC(LoRaWAN_Data, MIC, LoRaWAN_Data_Length, _frame_counter, Direction);
108 
109  // Load MIC in package
110  for(i = 0; i < 4; i++) {
111  LoRaWAN_Data[i + LoRaWAN_Data_Length] = MIC[i];
112  }
113  LoRaWAN_Data_Length += 4;
114 
115  // Update frame counter
116  ++_frame_counter;
117 
118  // Send Package
119  return AllWize::send(LoRaWAN_Data, LoRaWAN_Data_Length);
120 
121 }

336 
337  }

469 void AllWize_LoRaWAN::AES_Encrypt(uint8_t *Data, const uint8_t *Key) {
470 
471  uint8_t Row, Column, Round = 0;
472  uint8_t Round_Key[16];
473  uint8_t State[4][4];
474 
475  // Copy input to State arry
476  for (Column = 0; Column < 4; Column++) {
477  for (Row = 0; Row < 4; Row++) {
478  State[Row][Column] = Data[Row + (Column << 2)];
479  }
480  }
481 
482  // Copy key to round key
483  memcpy(&Round_Key[0], &Key[0], 16);
484 
485  // Add round key
486  AES_Add_Round_Key(Round_Key, State);
487 
488  // Preform 9 full rounds with mixed collums
489  for (Round = 1; Round < 10; Round++) {
490 
491  // Perform Byte substitution with S table
492  for (Column = 0; Column < 4; Column++) {
493  for (Row = 0; Row < 4; Row++) {
494  State[Row][Column] = AES_Sub_Byte(State[Row][Column]);
495  }
496  }
497 
498  // Perform Row Shift
499  AES_Shift_Rows(State);
500 
501  // Mix Collums
502  AES_Mix_Collums(State);
503 
504  // Calculate new round key
505  AES_Calculate_Round_Key(Round, Round_Key);
506 
507  // Add the round key to the Round_key
508  AES_Add_Round_Key(Round_Key, State);
509 
510  }
511 
512  // Perform Byte substitution with S table whitout mix collums
513  for (Column = 0; Column < 4; Column++) {
514  for (Row = 0; Row < 4; Row++) {
515  State[Row][Column] = AES_Sub_Byte(State[Row][Column]);
516  }
517  }
518 
519  // Shift rows
520  AES_Shift_Rows(State);
521 
522  // Calculate new round key
523  AES_Calculate_Round_Key(Round, Round_Key);
524 
525  // Add round key
526  AES_Add_Round_Key(Round_Key, State);
527 
528  // Copy the State into the data array
529  for (Column = 0; Column < 4; Column++) {
530  for (Row = 0; Row < 4; Row++) {
531  Data[Row + (Column << 2)] = State[Row][Column];
532  }
533  }

607 void AllWize_LoRaWAN::AES_Mix_Collums(uint8_t (*State)[4]) {
608 
609  uint8_t Row, Collum;
610  uint8_t a[4], b[4];
611 
612  for (Collum = 0; Collum < 4; Collum++) {
613  for (Row = 0; Row < 4; Row++) {
614  a[Row] = State[Row][Collum];
615  b[Row] = (State[Row][Collum] << 1);
616 
617  if ((State[Row][Collum] & 0x80) == 0x80) {
618  b[Row] ^= 0x1B;
619  }
620  }
621 
622  State[0][Collum] = b[0] ^ a[1] ^ b[1] ^ a[2] ^ a[3];
623  State[1][Collum] = a[0] ^ b[1] ^ a[2] ^ b[2] ^ a[3];
624  State[2][Collum] = a[0] ^ a[1] ^ b[2] ^ a[3] ^ b[3];

635 void AllWize_LoRaWAN::AES_Calculate_Round_Key(uint8_t Round, uint8_t *Round_Key) {
636 
637  uint8_t i, j, b, Rcon;
638  uint8_t Temp[4];
639 
640  //Calculate Rcon
641  Rcon = 0x01;
642  while (Round != 1) {
643  b = Rcon & 0x80;
644  Rcon = Rcon << 1;
645 
646  if (b == 0x80) {
647  Rcon ^= 0x1b;
648  }
649  Round--;
650  }
651 
652  // Calculate first Temp
653  // Copy laste byte from previous key and subsitute the byte, but shift the array contents around by 1.
654  Temp[0] = AES_Sub_Byte(Round_Key[12 + 1]);
655  Temp[1] = AES_Sub_Byte(Round_Key[12 + 2]);
656  Temp[2] = AES_Sub_Byte(Round_Key[12 + 3]);
657  Temp[3] = AES_Sub_Byte(Round_Key[12 + 0]);
658 
659  // XOR with Rcon
660  Temp[0] ^= Rcon;
661 
662  // Calculate new key
663  for (i = 0; i < 4; i++) {
664  for (j = 0; j < 4; j++) {
665  Round_Key[j + (i << 2)] ^= Temp[j];
666  Temp[j] = Round_Key[j + (i << 2)];
667  }
